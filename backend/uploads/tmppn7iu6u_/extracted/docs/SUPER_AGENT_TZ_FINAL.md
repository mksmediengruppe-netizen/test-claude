## 4.2 Cross-Chat Learning

**Система учится на ВСЕХ чатах ВСЕХ пользователей:**

```python
class CrossChatLearning:
    async def before_task(self, task: str, user_id: str):
        # 1. Найти похожие задачи из ВСЕХ чатов
        similar = await self.memory.search(
            query=task,
            scope="global",           # Все пользователи!
            include_errors=True,      # Включая ошибки
            include_successes=True,   # И успехи
            limit=10
        )

        # 2. Извлечь уроки
        lessons = []
        for s in similar:
            if s.had_error:
                lessons.append({
                    "type": "avoid",
                    "lesson": f"Не делать: {s.error_pattern}",
                    "correct": s.fixed_solution
                })
            if s.was_successful and s.rating > 4:
                lessons.append({
                    "type": "apply",
                    "lesson": f"Лучшая практика: {s.solution_pattern}"
                })

        # 3. Инжектить в контекст агента
        return self.build_context(lessons)
```

**Результат:** Агент не повторяет ошибки и использует лучшие решения.

## 6.2 Алгоритм оценки "Программист vs Агент"

```python
class TaskEstimator:
    """Оценка времени и стоимости для человека"""

    HUMAN_BENCHMARKS = {
        # task_type: (часы, ставка $/час)
        "landing_page":        (3.0, 25),   # $75
        "landing_complex":     (8.0, 25),   # $200
        "api_endpoint":        (2.0, 25),   # $50
        "bug_fix_simple":      (0.5, 25),   # $12.5
        "bug_fix_complex":     (4.0, 25),   # $100
        "react_component":     (1.5, 25),   # $37.5
        "full_page":           (4.0, 25),   # $100
        "database_migration":  (2.0, 25),   # $50
        "n8n_workflow":        (1.0, 25),   # $25
        "parser_script":       (1.5, 25),   # $37.5
        "design_review":       (0.5, 25),   # $12.5
    }

    async def compare(self, task: str, agent_cost: float, agent_time_sec: float):
        # Классифицируем задачу
        task_type = await self.classifier.classify(task)

        hours, rate = self.HUMAN_BENCHMARKS.get(task_type, (2.0, 25))
        human_cost = hours * rate
        human_time_min = hours * 60
        agent_time_min = agent_time_sec / 60

        return {
            "human_time": f"{hours} ч",
            "human_cost": f"${human_cost:.0f}",
            "agent_time": f"{agent_time_min:.1f} мин",
            "agent_cost": f"${agent_cost:.2f}",
            "time_speedup": f"{human_time_min / agent_time_min:.0f}x",
            "cost_savings": f"{(1 - agent_cost/human_cost) * 100:.0f}%"
        }
```

## 16.3 Реализация (React)

```typescript
// hooks/useTranslation.ts
import { useLanguageStore } from '@/stores/languageStore';
import ru from '@/i18n/locales/ru.json';
import en from '@/i18n/locales/en.json';

const locales = { ru, en };

export function useTranslation() {
  const { language } = useLanguageStore();

  const t = (key: string) => {
    const keys = key.split('.');
    let value = locales[language];
    for (const k of keys) {
      value = value?.[k];
    }
    return value || key;
  };

  return { t, language };
}

// Использование в компоненте
function ChatInput() {
  const { t } = useTranslation();
  return <input placeholder={t('chat.placeholder')} />;
}
```

function detectLanguage(): 'ru' | 'en' {
  // 1. Проверить сохранённый выбор
  const saved = localStorage.getItem('language');
  if (saved) return saved as 'ru' | 'en';

  // 2. Проверить язык браузера
  const browserLang = navigator.language.slice(0, 2);
  if (browserLang === 'ru') return 'ru';

  return 'ru';

# services/export_service.py
import zipfile
import io
from pathlib import Path

class ExportService:
    async def export_zip(self, chat_id: str) -> bytes:
        """Экспорт всех файлов чата в ZIP"""
        files = await self.get_chat_files(chat_id)

        buffer = io.BytesIO()
        with zipfile.ZipFile(buffer, 'w', zipfile.ZIP_DEFLATED) as zf:
            for file in files:
                zf.writestr(file.name, file.content)

        return buffer.getvalue()

    async def export_github(self, chat_id: str, repo: str, branch: str = "main"):
        """Push файлов в GitHub репозиторий"""
        files = await self.get_chat_files(chat_id)

        for file in files:
            await self.github.create_or_update_file(
                repo=repo,
                path=file.name,
                content=file.content,
                message=f"Add {file.name} from Super Agent",
                branch=branch
            )

    async def export_gist(self, chat_id: str, filename: str) -> str:
        """Создать GitHub Gist"""
        file = await self.get_chat_file(chat_id, filename)

        gist = await self.github.create_gist(
            description=f"Code from Super Agent - {filename}",
            files={filename: {"content": file.content}},
            public=False
        )

        return gist.html_url
```

# services/version_service.py
import difflib

class VersionService:
    async def save_version(self, chat_id: str, filename: str, content: str, message_id: str):
        """Сохранить новую версию файла"""
        # Получить текущую версию
        current = await self.get_current_version(chat_id, filename)
        new_version = (current.version + 1) if current else 1

        # Вычислить diff
        diff = None
        if current:
            diff = self.compute_diff(current.content, content)

        # Сохранить
        version = FileVersion(
            id=generate_id(),
            chat_id=chat_id,
            filename=filename,
            content=content,
            version=new_version,
            created_at=datetime.utcnow(),
            message_id=message_id,
            diff_from_prev=diff
        )

        await self.db.save(version)
        return version

    def compute_diff(self, old: str, new: str) -> str:
        """Вычислить unified diff"""
        diff = difflib.unified_diff(
            old.splitlines(keepends=True),
            new.splitlines(keepends=True),
            fromfile='previous',
            tofile='current'
        )
        return ''.join(diff)

    async def rollback(self, chat_id: str, filename: str, to_version: int):
        """Откатить к предыдущей версии"""
        version = await self.get_version(chat_id, filename, to_version)

        # Создать новую версию с содержимым старой
        return await self.save_version(
            chat_id=chat_id,
            filename=filename,
            content=version.content,
            message_id=f"rollback-to-v{to_version}"
        )

    async def get_history(self, chat_id: str, filename: str) -> List[FileVersion]:
        """Получить историю версий файла"""
        return await self.db.query(
            "SELECT * FROM file_versions WHERE chat_id = ? AND filename = ? ORDER BY version DESC",
            [chat_id, filename]
        )
```
